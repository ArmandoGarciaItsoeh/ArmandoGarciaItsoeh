import numpy as np import matplotlib.pyplot as plt

Matriz Original letra "E"

E = np.array([(0,0,1), (3,0,1), (3,1,1), (1,1,1), (1,2,1), (2,2,1), (2,3,1), (1,3,1), (1,4,1), (3,4,1), (3,5,1), (0,5,1), (0,0,1) ]) print("\nMatriz E Original\n") print(E)

Matriz de Reflexión en eje X

MRefX = np.array([(-1, 0, 0), ( 0, 1, 0), ( 0, 0, 1) ]) E_Reflejada = E.dot(MRefX) print("\nMatriz de Reflexión en eje X\n") print(MRefX) print("\nMatriz E reflejada\n") print(E_Reflejada)
