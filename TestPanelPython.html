packages = [ "https://cdn.holoviz.org/panel/0.14.3/dist/wheels/bokeh-2.4.3-py3-none-any.whl", "numpy", "matplotlib", "panel==0.14.1" ] import panel as pn import numpy as np import matplotlib.pyplot as plt xdata = np.array( [2.1,2.7,2.8,3.2,3.2,4.5,4.6,7.2,8.3,8.9,10,11.5,11.7,12.8,13.8,14.6,14.9] ) ydata = np.array( [2.94,4.32,1.12,0.64,1.92,3.6,8.28,1.44,9.96,3.56,20,11.5,11.7,17.92,22.08,23.36,17.88] ) x = np.linspace(xdata.min(), xdata.max(), 20) b = 0 m = 0 y = m*x + b fig, ax = plt.subplots() ax.scatter(xdata,ydata) ax.plot(x,y,color='red') display(plt, target="graph-area", append=False) slider = pn.widgets.FloatSlider(start=-3, end=3, name='m') sliderB = pn.widgets.FloatSlider(start=-5, end=20, name='b') def callback(new): m = new y = m*x + b ax.clear() ax.scatter(xdata,ydata) ax.plot(x,y,color='red') display(plt, target="graph-area", append=False) return f'y: {m} x + {b}' def callbackB(new2, pm): b = new2 y = pm*x + b ax.clear() ax.scatter(xdata,ydata) ax.plot(x,y,color='red') display(plt, target="graph-area", append=False) return f'y: {pm} x + {b}' pn.Row(slider, pn.bind(callback, slider)).servable(target='simple_app'); pn.Row(sliderB, pn.bind(callbackB, new2 = sliderB, pm=m)).servable(target='simple_app');
