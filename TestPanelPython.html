packages = [ "https://cdn.holoviz.org/panel/0.14.3/dist/wheels/bokeh-2.4.3-py3-none-any.whl", "numpy", "panel==0.14.1" ] plugins = [ "https://pyscript.net/latest/plugins/python/py_tutor.py" ] import panel as pn import numpy as np import matplotlib.pyplot as plt x = np.linspace(-25, 25, 50) b = 2 m = 1 y = m*x + b plt.plot(x,y) display(plt, target="graph-area", append=False) slider = pn.widgets.FloatSlider(start=0, end=10, name='Amplitude') def callback(new): b = new print(b) y = m*x + b plt.plot(x,y) display(plt, target="graph-area", append=False) return f'Amplitude is: {new}' pn.Row(slider, pn.bind(callback, slider)).servable(target='simple_app');
